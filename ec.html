03 スプレッドシートへ送信

▸ 初回セットアップ手順(Apps Scriptの作り方)
  1. 対象のGoogleスプレッドシートを開く
  2. メニュー「拡張機能」→「Apps Script」を開く
  3. デフォルトのコードを全て消し、下のコードを貼り付ける
  4. 右上「デプロイ」→「新しいデプロイ」
  5. 種類の選択(歯車)で「ウェブアプリ」を選ぶ
  6. 「アクセスできるユーザー」を 全員 にして「デプロイ」
  7. 発行された URL (.../exec) をコピーし、上の欄に貼り付けて保存
function doPost(e) {
  var SS_ID = '1gnioiJx6KHy0YnsG1Bg5DS5BRGUfx7IKzNmwltiNgxs';
  var data = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.openById(SS_ID);
  var sheet = ss.getSheetByName(data.sheetName);
  if (!sheet) {
    return ContentService.createTextOutput(JSON.stringify({success:false,error:'シートが見つかりません: '+data.sheetName}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // E列(購入者名)が空の最初の行を、5行目から探す
  var startRow = 5;
  var lastRow = Math.max(sheet.getLastRow(), startRow) + 1;
  var targetRow = -1;
  for (var r = startRow; r <= lastRow; r++) {
    var v = sheet.getRange(r, 5).getValue(); // E列
    if (!v) { targetRow = r; break; }
  }

  sheet.getRange(targetRow, 1).setValue(data.tantou || '');      // A 担当
  sheet.getRange(targetRow, 5).setValue(data.namae || '');       // E 購入者名
  sheet.getRange(targetRow, 6).setValue('様');                   // F
  sheet.getRange(targetRow, 7).setValue(data.ippan ? 1 : '');    // G 一般
  sheet.getRange(targetRow, 8).setValue(data.vip ? 1 : '');      // H VIP
  sheet.getRange(targetRow, 9).setValue(data.gohankai ? 1 : ''); // I ご飯会参加
  sheet.getRange(targetRow, 10).setValue(data.allergy || '');    // J アレルギー
  sheet.getRange(targetRow, 11).setValue(data.drink || '');      // K ドリンク
  sheet.getRange(targetRow, 13).setValue(data.kingaku || 0);     // M 小計金額

  return ContentService.createTextOutput(JSON.stringify({success:true,row:targetRow}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {

※ A/E/F/G/H/I/J/K/M列への書き込みは、現在のシート構成から推定したものです。 実際の列とズレている場合は、コード内の getRange(targetRow, 列番号) の数字を 実物のシートに合わせて直してください(A=1, B=2, C=3…の順)。
